    function createBird(vBirdType,vBirdPrice,
		vBirdsAdded)
    {
		var vBird = {
		birdType:"",
		birdPrice:"",
		birdsAvailable:0,
		birdsSoldOut:0}
		
		
      	vBird.birdType=vBirdType;
		vBird.birdPrice=vBirdPrice;
		vBird.birdsAvailable=vBird.birdsAvailable+vBirdsAdded;
		
		return vBird;
		
    }
	
	function addBirdToShop(vShop, vBird)
    {
		var birdExist = false;
		for(i = 0; i < vShop.length; i++) {
			var Bird = vShop[i];
		if (Bird.birdType==vBird.birdType){
			Bird.birdsAvailable=Bird.birdsAvailable+vBird.birdsAvailable;
			Bird.birdPrice=vBird.birdPrice;
			birdExist = true;
			}
		
	    }
		if (birdExist == false){ 
				vShop.push(vBird);
				
			}
	}
	
	function sellBird(vShop, birdType, vCount)
    {
		var birdExist = false;
		for(i = 0; i < vShop.length; i++) {
			var Bird = vShop[i];
		if (Bird.birdsAvailable < vCount){ 
		console.log("CAN'T SELL BIRDS WITH TYPE: "+birdType +". NUMBER OF BIRDS IS NOT ENOUGH. YOU WANT TO SELL "+ vCount + " YOU GOT " + Bird.birdsAvailable );
				return;
			}
		if (Bird.birdType==birdType){
			Bird.birdsAvailable=Bird.birdsAvailable-vCount;
			Bird.birdsSoldOut=Bird.birdsSoldOut+vCount;
			birdExist = true;
			}
		
	    }
		if (birdExist == false){ 
		console.log("CAN'T SELL BIRDS WITH TYPE: "+birdType +". NO SUCH TYPES IN THE SHOP");
				
			}
	}
